Build a quadrant assignment function based on the validated position mapping from Task 1:
```typescript
function getQuadrant(round: number, position: number): 'TL' | 'TR' | 'BL' | 'BR' | 'connecting' {
  if (round > 4) return 'connecting' // Rounds 5-6 are semifinals/finals
  // Trace back to round 1 position
  let r1Position = position
  for (let r = round; r > 1; r--) {
    r1Position = (r1Position - 1) * 2 + 1 // First feeder position
  }
  // r1Position is now the FIRST round-1 position that feeds into this matchup
  if (r1Position <= 8) return 'TL'
  if (r1Position <= 16) return 'TR'
  if (r1Position <= 24) return 'BL'
  return 'BR'
}
```

Add a comment documenting the position ranges derived from Task 1 validation.

2. **Filter and normalize matchups per quadrant.** Each quadrant gets matchups from rounds 1-4 only, with positions renormalized to 1-based within the quadrant:
```typescript
function normalizeQuadrant(quadrantMatchups: MatchupData[]): MatchupData[] {
  const byRound = new Map<number, MatchupData[]>()
  for (const m of quadrantMatchups) {
    const arr = byRound.get(m.round) || []
    arr.push(m)
    byRound.set(m.round, arr)
  }

  return quadrantMatchups.map((m) => {
    const roundMatchups = byRound.get(m.round)!
    const minPos = Math.min(...roundMatchups.map((r) => r.position))
    return { ...m, position: m.position - minPos + 1 }
  })
}
```

Helper to filter voteLabels to only matchups in a quadrant:
```typescript
function filterVoteLabels(
  labels: Record<string, Record<string, number>> | undefined,
  quadrantMatchups: MatchupData[]
): Record<string, Record<string, number>> | undefined {
  if (!labels) return undefined
  const ids = new Set(quadrantMatchups.map((m) => m.id))
  const filtered: Record<string, Record<string, number>> = {}
  for (const [k, v] of Object.entries(labels)) {
    if (ids.has(k)) filtered[k] = v
  }
  return Object.keys(filtered).length > 0 ? filtered : undefined
}
```
