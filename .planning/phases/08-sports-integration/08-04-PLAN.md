    ```typescript
    import { NextResponse } from 'next/server'
    import { getProvider } from '@/lib/sports/provider'
    import { getActiveSportsBrackets, syncBracketResults } from '@/lib/dal/sports'

    export async function GET(request: Request) {
      // 1. Authenticate with CRON_SECRET (Vercel injects this for cron jobs)
      const authHeader = request.headers.get('authorization')
      if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
        return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
      }

      try {
        // 2. Check if there are any active sports brackets to sync
        const activeBrackets = await getActiveSportsBrackets()
        if (activeBrackets.length === 0) {
          return NextResponse.json({ status: 'no_active_brackets', synced: 0 })
        }

        // 3. Adaptive gate: check if any games are in progress
        const provider = getProvider()
        const gamesInProgress = await provider.areGamesInProgress()

        if (!gamesInProgress) {
          // No games active -- do a lightweight schedule check at most once per hour
          // Check if any brackets haven't been synced in the last hour
          const staleThreshold = new Date(Date.now() - 60 * 60 * 1000)
          const staleBrackets = activeBrackets.filter(
            b => !b.lastSyncAt || b.lastSyncAt < staleThreshold
          )

          if (staleBrackets.length === 0) {
            return NextResponse.json({
              status: 'no_games_in_progress',
              synced: 0,
              activeBrackets: activeBrackets.length
            })
          }

          // Sync stale brackets with current schedule data
          // (picks up schedule changes, postponements, etc.)
        }

        // 4. Fetch today's games with updated scores
        const today = new Date()
        const dateStr = formatSportsDataIODate(today)
        const games = await provider.getGamesByDate(dateStr)

        // 5. Sync each active bracket
        let syncedCount = 0
        for (const bracket of activeBrackets) {
          try {
            await syncBracketResults(bracket.id, games)
            syncedCount++
          } catch (err) {
            console.error(`Failed to sync bracket ${bracket.id}:`, err)
            // Continue syncing other brackets
          }
        }

        return NextResponse.json({
          status: 'synced',
          synced: syncedCount,
          gamesInProgress,
          gamesChecked: games.length
        })
      } catch (err) {
        console.error('Sports sync cron error:', err)
        return NextResponse.json(
          { error: 'Sync failed', details: String(err) },
          { status: 500 }
        )
      }
    }

    // Format date for SportsDataIO API: "2026-MAR-20"
    function formatSportsDataIODate(date: Date): string {
      const months = ['JAN', 'FEB', 'MAR', 'APR', 'MAY', 'JUN',
                       'JUL', 'AUG', 'SEP', 'OCT', 'NOV', 'DEC']
      const year = date.getFullYear()
      const month = months[date.getMonth()]
      const day = date.getDate().toString().padStart(2, '0')
      return `${year}-${month}-${day}`
    }
    ```

    Key design decisions:
    - Per-bracket try/catch: one bracket failing doesn't block others (matching webhook per-operation pattern from 06-01)
    - Adaptive polling: early exit when no active brackets, lightweight check when no games in progress
    - `formatSportsDataIODate` helper formats dates for SportsDataIO API's required format
    - Returns JSON response with sync stats for monitoring
    - CRON_SECRET auth matches Vercel's built-in cron authentication

    **Vercel config** -- Create `vercel.json` at project root:
    ```json
    {
      "crons": [
        {
          "path": "/api/cron/sports-sync",
          "schedule": "*/2 * * * *"
        }
      ]
    }
    ```

    This runs the sync every 2 minutes. The endpoint itself gates on active games, so API load is minimal outside game times.

    **Environment variable note**: `CRON_SECRET` is automatically provided by Vercel for cron jobs. For local development, add `CRON_SECRET=local-dev-secret` to `.env.local`. Document this in the plan summary.
