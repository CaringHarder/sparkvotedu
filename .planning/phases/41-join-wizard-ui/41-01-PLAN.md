```typescript
const createWizardSchema = z.object({
  code: z.string().regex(/^\d{6}$/),
})

export async function createWizardParticipant(input: { code: string }): Promise<JoinResult> {
  const parsed = createWizardSchema.safeParse(input)
  if (!parsed.success) return { error: parsed.error.issues[0].message }
  const session = await findSessionByCode(parsed.data.code)
  if (!session) return { error: 'Invalid class code' }
  if (session.status === 'ended') return { session: { id: session.id, code: session.code, name: session.name, status: session.status, teacherName: session.teacher.name }, sessionEnded: true }
  const participant = await createParticipant(session.id, null, undefined, '')
  broadcastParticipantJoined(session.id).catch(() => {})
  return {
    participant: toParticipantData(participant),
    session: { id: session.id, code: session.code, name: session.name, status: session.status, teacherName: session.teacher.name },
    returning: false,
  }
}
```
