   ```
   function handleReopen(e: Event) {
     e.stopPropagation()
     startTransition(async () => {
       if (itemType === 'bracket') {
         const result = await reopenBracket({ bracketId: itemId })
         if ('success' in result && result.success) {
           onReopened?.()
         }
       } else {

         if ('success' in result && result.success) {
           onReopened?.()
         }
       }
     })
   }
   ```
