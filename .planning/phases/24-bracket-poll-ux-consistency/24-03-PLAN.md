   ```typescript
       // Get top 2 from standings for the reveal display

         if (b.wins !== a.wins) return b.wins - a.wins
         return b.pointDiff - a.pointDiff
       })
       const top1 = sortedStandings[0]?.entrantName ?? 'Finalist'
       const top2 = sortedStandings[1]?.entrantName ?? 'Finalist'

       const timer = setTimeout(() => {
         setRevealState({ entrant1Name: top1, entrant2Name: top2 })
       }, 2000)
       return () => clearTimeout(timer)
     }
   }, [bracketCompleted, revealState, showCelebration, standings])
   ```
