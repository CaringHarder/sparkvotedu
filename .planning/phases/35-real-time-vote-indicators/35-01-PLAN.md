    Add this export function after the existing poll DAL functions:
    ```typescript
    export async function getPollVoterParticipantIds(pollId: string): Promise<string[]> {
      const votes = await prisma.pollVote.findMany({
        where: { pollId, rank: 1 },
        select: { participantId: true },
        distinct: ['participantId'],
      })
      return votes.map((v) => v.participantId)
    }
    ```
    Using `rank: 1` ensures one row per voter (works for both simple polls where rank is always 1, and ranked polls where each voter has a rank=1 entry).
