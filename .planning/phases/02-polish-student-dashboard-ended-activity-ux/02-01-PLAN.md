```typescript
const CLOSED_STATUSES = new Set(['completed', 'closed'])
const activeActivities = activities.filter(a => !CLOSED_STATUSES.has(a.status))
const closedActivities = activities.filter(a => CLOSED_STATUSES.has(a.status))
```

**3. Auto-navigation guard:** Change the auto-nav useEffect (lines 43-52):
- Only auto-navigate when `activeActivities.length === 1` (not `activities.length === 1`)
- When there's 1 active + N closed, still auto-navigate to the active one
- When there are 0 active activities (all closed), do NOT auto-navigate — show grid instead
- Navigate using the single active activity, not `activities[0]`

**4. Empty state logic:** Change the `activities.length === 0` check (line 67):
- If `activities.length === 0` (truly no activities at all): show EmptyState as before
- If `activeActivities.length === 0 && closedActivities.length > 0`: render friendly "no active" message ABOVE the Closed section (do NOT use EmptyState component — render inline)

**5. Friendly "no active" message** (when all activities are closed):
```tsx
<div className="flex flex-col items-center py-8 text-center">
  <h3 className="text-lg font-medium text-muted-foreground">
    No active activities right now
  </h3>
  <p className="mt-1 text-sm text-muted-foreground/70">
    Hang tight — your teacher will start something soon!
  </p>
</div>
```

**6. Section divider** (rendered between active and closed sections, only when BOTH have items):
```tsx
<div className="flex items-center gap-3 py-2">
  <div className="h-px flex-1 bg-muted-foreground/20" />
  <span className="text-xs font-medium text-muted-foreground/60 uppercase tracking-wider">Closed</span>
  <div className="h-px flex-1 bg-muted-foreground/20" />
</div>
```

**7. Two-section layout with LayoutGroup:** Wrap both sections in `<LayoutGroup>` for cross-container layoutId animation. Structure:

```tsx
<div className="space-y-4">
  <h2 className="text-lg font-semibold">Brackets/Polls</h2>
  <LayoutGroup>
    {/* Active section */}
    {activeActivities.length > 0 && (
      <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
        <AnimatePresence mode="popLayout">
          {activeActivities.map((activity) => (
            <motion.div
              key={activity.id}
              layoutId={activity.id}
              layout
              initial={{ opacity: 0, scale: 0.95 }}
              animate={{ opacity: 1, scale: 1 }}
              exit={{ opacity: 0.55 }}
              transition={{ duration: 0.3, ease: 'easeInOut' }}
            >
              <ActivityCard activity={activity} isClosed={false} onClick={...} />
            </motion.div>
          ))}
        </AnimatePresence>
      </div>
    )}

    {/* "No active" message when all closed */}
    {activeActivities.length === 0 && closedActivities.length > 0 && (
      <friendly-message />
    )}

    {/* Divider when both sections have items */}
    {activeActivities.length > 0 && closedActivities.length > 0 && (
      <section-divider />
    )}

    {/* Closed section */}
    {closedActivities.length > 0 && (
      <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
        <AnimatePresence mode="popLayout">
          {closedActivities.map((activity) => (
            <motion.div
              key={activity.id}
              layoutId={activity.id}
              layout
              animate={{ opacity: 0.55 }}
              transition={{ duration: 0.3, ease: 'easeInOut' }}
            >
              <ActivityCard activity={activity} isClosed={true} onClick={...} />
            </motion.div>
          ))}
        </AnimatePresence>
      </div>
    )}
  </LayoutGroup>
</div>
```

**8. onClick handler for closed cards:** Same navigation logic as active cards — tapping a closed card navigates to the existing read-only view (bracket results or poll closed page). Reuse the same onClick handler.
